# Examples

See real-world examples of using Harukit components.

## Basic Examples

### Simple Form

A basic form with input fields and buttons.

```tsx
import { Button } from "@/components/button"
import { Input } from "@/components/input"
import { Label } from "@/components/label"
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/card"

export default function ContactForm() {
  return (
    <Card className="w-full max-w-md mx-auto">
      <CardHeader>
        <CardTitle>Contact Us</CardTitle>
        <CardDescription>
          Send us a message and we'll get back to you as soon as possible.
        </CardDescription>
      </CardHeader>
      <CardContent className="space-y-4">
        <div className="space-y-2">
          <Label htmlFor="name">Name</Label>
          <Input id="name" placeholder="Enter your name" />
        </div>
        <div className="space-y-2">
          <Label htmlFor="email">Email</Label>
          <Input id="email" type="email" placeholder="Enter your email" />
        </div>
        <div className="space-y-2">
          <Label htmlFor="message">Message</Label>
          <Input id="message" placeholder="Enter your message" />
        </div>
        <Button className="w-full">Send Message</Button>
      </CardContent>
    </Card>
  )
}
```

### Navigation Bar

A simple navigation bar with buttons and tooltips.

```tsx
import { Button } from "@/components/button"
import { Tooltip, TooltipContent, TooltipProvider, TooltipTrigger } from "@/components/tooltip"

export default function NavigationBar() {
  return (
    <nav className="flex items-center justify-between p-4 bg-background border-b">
      <div className="flex items-center space-x-4">
        <h1 className="text-xl font-bold">My App</h1>
      </div>
      
      <div className="flex items-center space-x-2">
        <TooltipProvider>
          <Tooltip>
            <TooltipTrigger asChild>
              <Button variant="ghost" size="sm">
                Home
              </Button>
            </TooltipTrigger>
            <TooltipContent>
              <p>Go to home page</p>
            </TooltipContent>
          </Tooltip>
          
          <Tooltip>
            <TooltipTrigger asChild>
              <Button variant="ghost" size="sm">
                Settings
              </Button>
            </TooltipTrigger>
            <TooltipContent>
              <p>Open settings</p>
            </TooltipContent>
          </Tooltip>
        </TooltipProvider>
        
        <Button>Sign In</Button>
      </div>
    </nav>
  )
}
```

### FAQ Section

An FAQ section using the accordion component.

```tsx
import { Accordion, AccordionContent, AccordionItem, AccordionTrigger } from "@/components/accordion"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/card"

export default function FAQSection() {
  return (
    <Card>
      <CardHeader>
        <CardTitle>Frequently Asked Questions</CardTitle>
      </CardHeader>
      <CardContent>
        <Accordion type="single" collapsible className="w-full">
          <AccordionItem value="item-1">
            <AccordionTrigger>What is Harukit?</AccordionTrigger>
            <AccordionContent>
              Harukit is a modern CLI tool for adding beautiful, accessible UI components to your React projects. 
              It works like shadcn/ui - components are copied to your project, giving you full control over their code and styling.
            </AccordionContent>
          </AccordionItem>
          
          <AccordionItem value="item-2">
            <AccordionTrigger>How do I install Harukit?</AccordionTrigger>
            <AccordionContent>
              You can install Harukit by running <code>npx harukit@latest init</code> in your project directory. 
              This will set up the necessary configuration and install all required dependencies.
            </AccordionContent>
          </AccordionItem>
          
          <AccordionItem value="item-3">
            <AccordionTrigger>Is Harukit free to use?</AccordionTrigger>
            <AccordionContent>
              Yes, Harukit is completely free and open source. You can use it in any project, 
              commercial or personal, without any restrictions.
            </AccordionContent>
          </AccordionItem>
        </Accordion>
      </CardContent>
    </Card>
  )
}
```
